PPrivate Sub Worksheet_BeforeDoubleClick(ByVal Target As Range, Cancel As Boolean)

    ' 対象セル：AD4 と AE4
    If Not Intersect(Target, Range("AD4, AE4")) Is Nothing Then
        
        Cancel = True  ' 編集モードを無効化

        ' ◎ が入っていれば消す、空なら ◎ を入れる
        If Target.Value = "◎" Then
            Target.ClearContents
        Else
            Target.Value = "◎"
        End If

    End If

End Sub

Private Sub Worksheet_SelectionChange(ByVal Target As Range)

    Dim watchRange As Range
    Set watchRange = Me.Range("AC8:AC73")

    If Target.Cells.Count > 1 Then Exit Sub

    ' E4 が選択されたら openingcheck を表示
    If Not Application.Intersect(Target, Me.Range("E4")) Is Nothing Then
        openingcheck.Show vbModal
        Exit Sub
    End If

    ' AC8:AC73 が選択されたら TenKey を表示
    If Not Application.Intersect(Target, watchRange) Is Nothing Then
        TenKey.Show vbModal
    End If

End Sub
